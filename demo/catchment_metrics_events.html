Introduction¶

Taking the start and end time of each event and calculating my own indices using INCA, SPARTACUS, and other station data.

Calculate catchment characteristics before and during event. Here there are many options to explore both with station observations and gridded datasets.

  • precipitation intensity
  • event precipitation
  • antecedent precipitation
  • antecedent temperature
  • snow covered area
  • glaciated area(?)
  • ACDA

def get_event_raster(start, end, raster_timeseries):
    event_raster = raster_timeseries.sel(time=slice(start, end))
    return event_raster


def total_event_precip(precip_timeseries, start, end):
    event_precip_series = watersedimentpulses.events.metrics.get_event_data(
        start, end, precip_timeseries
    )
    if len(event_precip_series) < 1:
        precip_sum = np.nan
    else:
        precip_sum = event_precip_series.sum()
    return precip_sum


def max_event_precip_intensity(precip_timeseries, start, end):
    event_precip_series = watersedimentpulses.events.metrics.get_event_data(
        start, end, precip_timeseries
    )
    if len(event_precip_series) < 1:
        precip_int = np.nan
    else:
        precip_int = event_precip_series.max()
    return precip_int


def total_event_precip_raster(precip_timeseries_raster, start, end):
    event_precip_raster = get_event_raster(start, end, precip_timeseries_raster)
    attrs = event_precip_raster.attrs.copy()
    attrs.update(
        dict(
            units="mm",
            long_name="event precipitation sum",
            description="precipitation sum of event",
        )
    )
    if len(event_precip_raster.time) < 1:
        # this will return an array of NaNs, skipna must be set to true or else an array of zeros is returned
        precip_sum = event_precip_raster.sum(dim="time", skipna=True, min_count=1)
    else:
        precip_sum = event_precip_raster.sum(dim="time", skipna=False, min_count=1)
    return precip_sum.rename("event_precip_sum").assign_attrs(attrs)


def max_event_precip_intensity_raster(precip_timeseries_raster, start, end):
    event_precip_raster = get_event_raster(start, end, precip_timeseries_raster)
    freq = pd.infer_freq(precip_timeseries_raster.time)
    attrs = event_precip_raster.attrs.copy()
    attrs.update(
        dict(
            units=f"mm/{freq.lower()}",
            long_name="maximum precipitation intensity",
            description="maximum precipitation intensity of event",
        )
    )
    if len(event_precip_raster.time) < 1:
        # this will return an array of NaNs, skipna must be set to true or else an array of zeros is returned
        precip_sum = event_precip_raster.sum(dim="time", skipna=True, min_count=1)
    else:
        precip_sum = event_precip_raster.max(dim="time", skipna=False)

    return precip_sum.rename("event_precip_sum").assign_attrs(attrs)

def antecedent_start(start, days_before):
    timepad = dateutil.relativedelta.relativedelta(days=days_before)
    ap_start = pd.to_datetime(start) - timepad
    return pd.to_datetime(ap_start.date())


def antecedent_precip_sum(start, days_before, precip_timeseries):
    ap_series = watersedimentpulses.events.metrics.get_event_data(
        antecedent_start(start, days_before), start, precip_timeseries
    )
    return ap_series.sum(min_count=days_before)

def NAPI(ap_series, k):
    """Normalised antecedent precipitation index (Heggen, 2001)"""
    api = []
    t_range = np.arange(0, -len(ap_series), -1)
    for t in t_range:
        i = t - 1
        P_t = ap_series.iloc[i]
        api.append(P_t * k**-t)
    return np.array(api).sum() / (k**-t_range).sum()


def norm_antecedent_precip_index(precip_timeseries, start, d, k=0.9):
    ap_series = precip_timeseries[antecedent_start(event.start, d) : event.start]
    if len(ap_series) < d:
        return np.nan
    freq = pd.infer_freq(ap_series.index)
    if freq != "D":
        ap_series = ap_series.resample("D", offset="7H").sum().resample("D").sum()
    return NAPI(ap_series, k=k)

def event_global_radiation(
    start, end, globrad_ts, start_hour_threshold=15, end_hour_threshold=9
):
    start = pd.to_datetime(start)
    end = pd.to_datetime(end)

    # get the timeseries of dates of event
    full_ts = watersedimentpulses.events.metrics.get_event_data(
        start.date(), end.date(), globrad_ts
    )

    # if event covers only one date, use that day
    if len(full_ts) < 2:
        event_globrad = full_ts.mean().values[0]
    # if event covers two dates or longer
    else:
        start_threshold = start.date() + dateutil.relativedelta.relativedelta(
            hours=+start_hour_threshold
        )
        end_threshold = end.date() + dateutil.relativedelta.relativedelta(
            hours=+end_hour_threshold
        )
        # if the event starts and ends before thresholds
        if start <= start_threshold and end <= end_threshold:
            # do not include last date in average
            event_globrad = full_ts.iloc[0:-1].mean().values[0]
        # else if the event starts and ends after thresholds
        elif start > start_threshold and end > end_threshold:
            # do not include first day in event
            event_globrad = full_ts.iloc[1:].mean().values[0]
        # else if event starts before start_threshold and ends after end_threshold
        elif start <= start_threshold and end > end_threshold:
            # take average of all dates
            event_globrad = full_ts.mean().values[0]
        # else if event starts after start_threshold and ends before end_threshold
        elif start > start_threshold and end <= end_threshold:
            # exclude first and last day from average
            event_globrad = full_ts.iloc[1:-1].mean().values[0]
            if len(full_ts.iloc[1:-1]) < 1:
                warnings.warn(
                    "Event starts after noon and ends before noon on the following day. Therefore an empty slice of daily data was selected for the average. The value will be NaN."
                )
                print(start, end)
    return event_globrad

def animate_event_raster(
    raster, time_dim="time", x_dim="lon", y_dim="lat", cmap="PuBu"
):
    fig, ax = plt.subplots()

    def first_frame():
        raster.sel(time=raster[time_dim][0]).plot(
            ax=ax,
            x=x_dim,
            y=y_dim,
            vmin=raster.min(),
            vmax=raster.max(),
            add_colorbar=True,
            cmap=cmap,
        )
        ax.set_title("")

    def animate_event(time):
        plt.cla()
        raster.sel(time=time).plot(
            ax=ax,
            x=x_dim,
            y=y_dim,
            vmin=raster.min(),
            vmax=raster.max(),
            add_colorbar=False,
            cmap=cmap,
        )
        plt.title(f"{pd.to_datetime(time.values):%Y-%m-%d %H:%M}")

    ani = FuncAnimation(
        fig,
        animate_event,
        frames=raster[time_dim],
        init_func=first_frame(),
        repeat=False,
    )
    return ani

def antecedent_snowcover_change(snowcover, start, end, days_before=1, var="SCA"):
    start = pd.to_datetime(start)
    end = pd.to_datetime(end)
    ts = snowcover.loc[antecedent_start(start, days_before) : end][var]
    if len(ts) == 0:
        return np.nan
    else:
        return ts.iloc[-1] - ts.iloc[0]

def antecedent_freezingIndex(start, days_before, temp_timeseries):
    ap_series = temp_timeseries[

def antecendent_global_radiation(start,end,globrad_ts,days_before = 5,start_hour_threshold = 15):
    # convert to datetime objects
    start = pd.to_datetime(start)
    end = pd.to_datetime(end)
    # calculate start threshold time for start date
    start_threshold = start.date() + dateutil.relativedelta.relativedelta(
        hours=+start_hour_threshold
    )
    # get the "full" timeseries of antecedent and event start date
    full_ts = watersedimentpulses.events.metrics.get_event_data(
        antecedent_start(start, days_before), start.date(), globrad_ts
    )
    
    # if the event start is earlier than the event start threshold 
    if start <= start_threshold:
        # exclude the event start date from average
        antecendent_globrad = full_ts.iloc[:-1].mean().values[0]
    else:
        # include start date in average
        antecendent_globrad = full_ts.mean().values[0]
    return antecendent_globrad
